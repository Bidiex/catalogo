<script>
const headerContent = document.getElementById('headerContent')
const logo = document.getElementById('logo')
const name = document.getElementById('businessName')
const description = document.getElementById('headerDescription')

let isCompact = false

// ðŸ”¥ TIMELINE ÃšNICO
const tl = gsap.timeline({
  paused: true,
  defaults: { duration: 0.28, ease: "power2.out" }
})

// ESTADO COMPACTO
tl.to(headerContent, {
  padding: "10px 16px",
  gap: 6
}, 0)

.to(logo, {
  width: 36,
  height: 36,
  borderRadius: 10
}, 0)

.to(name, {
  fontSize: "1.05rem"
}, 0)

.to(description, {
  opacity: 0,
  height: 0,
  marginTop: 0
}, 0)

// ðŸ‘‰ DISPARO ÃšNICO AL PRIMER SCROLL
window.addEventListener(
  'scroll',
  () => {
    if (!isCompact && window.scrollY > 0) {
      isCompact = true
      tl.play()
    }

    if (isCompact && window.scrollY === 0) {
      isCompact = false
      tl.reverse()
    }
  },
  { passive: true }
)
</script>
